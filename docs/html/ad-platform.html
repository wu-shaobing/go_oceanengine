package main

import (
    "context"
    "fmt"
    ad_open_sdk_go "github.com/oceanengine/ad_open_sdk_go"
    "github.com/oceanengine/ad_open_sdk_go/api"
    "github.com/oceanengine/ad_open_sdk_go/config"
    "github.com/oceanengine/ad_open_sdk_go/models"
)

func main() {
    // 1. 创建 SDK 配置
    conf := &config.Configuration{
        Host:      "open.oceanengine.com",
        Scheme:    "https",
        LogEnable: true,
    }

    // 2. 创建 API 客户端
    client := ad_open_sdk_go.NewAPIClient(conf)

    // 3. 设置认证
    ctx := context.WithValue(
        context.Background(),
        config.ContextAccessToken,
        "YOUR_ACCESS_TOKEN",
    )

    // 4. 构建请求参数
    req := api.NewProjectListV30Request()
    req.AdvertiserId = 1234567890
    req.Filtering = &models.ProjectListV30Filtering{
        Status: "ENABLE",  // 筛选启用的项目
    }
    req.Page = 1
    req.PageSize = 10

    // 5. 发起 API 请求
    resp, httpResp, err := client.ProjectApi.ListV30(ctx, req)
    if err != nil {
        fmt.Printf("请求失败: %v\n", err)
        return
    }

    // 6. 处理响应
    if resp.Code != 0 {
        fmt.Printf("API 错误: %s (code: %d)\n", resp.Message, resp.Code)
        return
    }

    fmt.Printf("总计 %d 个项目\n", resp.Data.PageInfo.TotalNumber)
    for _, project := range resp.Data.List {
        fmt.Printf("- ID: %d, 名称: %s, 状态: %s\n",
            project.ProjectId,
            project.Name,
            project.Status,
        )
    }
}

package main

import (
    "context"
    "fmt"
    "log"
    "os"
    
    ad_open_sdk_go "github.com/oceanengine/ad_open_sdk_go"
    "github.com/oceanengine/ad_open_sdk_go/config"
)

func UploadImage(advertiserID int64, imagePath string) (string, error) {
    // 1. 初始化 SDK
    configuration := config.NewConfiguration()
    apiClient := ad_open_sdk_go.Init(configuration)
    
    // 2. 设置 Access Token
    ctx := context.WithValue(
        context.Background(),
        config.ContextAccessToken,
        "YOUR_ACCESS_TOKEN",
    )
    
    // 3. 读取文件
    file, err := os.Open(imagePath)
    if err != nil {
        return "", fmt.Errorf("读取文件失败: %w", err)
    }
    defer file.Close()
    
    // 4. 构建请求
    req := apiClient.FileImageAdApi().Post(ctx)
    req.AdvertiserId(advertiserID)
    req.ImageFile(file)  // 传入文件句柄
    req.ImageSignature("")  // 可选：文件MD5签名
    
    // 5. 执行上传
    resp, httpResp, err := req.Execute()
    if err != nil {
        return "", fmt.Errorf("上传失败: %w", err)
    }
    
    // 6. 检查响应
    if resp.Code != 0 {
        return "", fmt.Errorf("API错误: %s (code: %d)", resp.Message, resp.Code)
    }
    
    // 7. 返回素材 ID
    imageID := resp.Data.ImageId
    fmt.Printf("上传成功！素材ID: %s\n", imageID)
    
    return imageID, nil
}

func main() {
    advertiserID := int64(1234567890)
    imagePath := "./banner.jpg"
    
    imageID, err := UploadImage(advertiserID, imagePath)
    if err != nil {
        log.Fatal(err)
    }
    
    fmt.Printf("图片已上传，ID: %s\n", imageID)
}

// 进度监控 Reader
type ProgressReader struct {
    reader    io.Reader
    total     int64
    current   int64
    onProgress func(current, total int64)
}

func (pr *ProgressReader) Read(p []byte) (int, error) {
    n, err := pr.reader.Read(p)
    pr.current += int64(n)
    if pr.onProgress != nil {
        pr.onProgress(pr.current, pr.total)
    }
    return n, err
}

import "time"

func UploadWithRetry(uploadFunc func() error, maxRetries int) error {
    var err error
    for i := 0; i < maxRetries; i++ {
        err = uploadFunc()
        if err == nil {
            return nil  // 成功
        }
        
        // 失败后等待后重试
        if i < maxRetries-1 {
            waitTime := time.Duration(i+1) * time.Second
            fmt.Printf("上传失败，%v 后重试 (%d/%d)\n", waitTime, i+1, maxRetries)
            time.Sleep(waitTime)
        }
    }
    return fmt.Errorf("上传失败，已重试 %d 次: %w", maxRetries, err)
}

1. 分批处理大量数据

func BatchProcessInChunks(allIDs []int64, chunkSize int) {
    // 分批处理，避免一次性处理太多
    for i := 0; i < len(allIDs); i += chunkSize {
        end := i + chunkSize
        if end > len(allIDs) {
            end = len(allIDs)
        }
        
        chunk := allIDs[i:end]
        fmt.Printf("\n处理第 %d-%d 个对象...\n", i+1, end)
        
        // 处理当前批次
        results := BatchUpdateStatus(advertiserID, chunk, "PAUSE")
        
        // 等待一下，避免频繁请求
        time.Sleep(2 * time.Second)
    }
}

2. 错误重试机制

func BatchWithRetry(ids []int64, maxRetries int) {
    failedIDs := ids
    
    for retry := 0; retry < maxRetries && len(failedIDs) > 0; retry++ {
        if retry > 0 {
            fmt.Printf("\n第 %d 次重试，剩余 %d 个失败对象...\n", retry, len(failedIDs))
            time.Sleep(time.Duration(retry) * time.Second)
        }
        
        results := BatchUpdateStatus(advertiserID, failedIDs, "ENABLE")
        
        // 收集失败的 ID
        newFailedIDs := make([]int64, 0)
        for _, r := range results {
            if !r.Success {
                newFailedIDs = append(newFailedIDs, r.AdID)
            }
        }
        
        failedIDs = newFailedIDs
    }
    
    if len(failedIDs) > 0 {
        fmt.Printf("\n仍有 %d 个对象操作失败: %v\n", len(failedIDs), failedIDs)
    }
}

❓ 常见问题

❓ 批量操作的最大数量限制？

单次API调用通常支持 1-100 个对象。建议单次 20-50 个，超过 100 个应分批处理。

❓ 批量操作失败了怎么办？

• 检查错误信息，确认是网络问题还是参数错误
• 对失败的对象进行重试
• 降低并发数，增加批次间隔
• 分批处理，避免一次性操作太多

❓ 如何加速批量操作？

• 使用并发处理（Goroutine + Semaphore）
• 合理设置并发数，避免触发限流
• 使用批量接口（如果 API 支持）
• 复用 SDK 客户端，避免重复初始化

❓ 批量创建后如何验证？

• 记录创建的对象 ID
• 调用查询接口验证状态
• 导出结果到 JSON/CSV 文件便于检查
• 在广告管理后台人工检查重要对象

Go 代码示例

                        
package main

import (
    "fmt"
    "net/url"
)

func GenerateAuthURL(appID, redirectURI, state string) string {
    baseURL := "https://ad.oceanengine.com/openapi/audit/oauth.html"
    params := url.Values{}
    params.Add("app_id", appID)
    params.Add("state", state)  // 状态参数，用于防止CSRF攻击
    params.Add("scope", "[1,2,3]")  // 权限范围
    params.Add("redirect_uri", redirectURI)
    
    return fmt.Sprintf("%s?%s", baseURL, params.Encode())
}

func main() {
    appID := "YOUR_APP_ID"
    redirectURI := "https://your-domain.com/callback"
    state := "random_state_string_12345"  // 生成随机字符串
    
    authURL := GenerateAuthURL(appID, redirectURI, state)
    fmt.Println("授权链接：", authURL)
    
    // 引导用户访问该链接进行授权
}

                        ❓
                        常见问题 FAQ
                    

                    

                        

                            
❓ Token 有效期是多久？

                            
Access Token 默认有效期为 30 天，Refresh Token 有效期更长，具体以平台返回为准。

                        

                        

                            
❓ 如何判断 Token 是否过期？

                            
调用 API 时如果返回错误码 40101（token 失效），说明 token 已过期，需要刷新。建议提前刷新而不是等到过期。

                        

                        

                            
❓ 多个广告主如何管理 Token？

                            
每个广告主需要单独授权，获取各自的 access_token。建议使用 advertiser_id 作为 key 存储对应的 token。

                        

                        

                            
❓ Refresh Token 失效了怎么办？

                            
如果 refresh_token 也失效，需要重新引导用户进行 OAuth 授权，获取新的 authorization code。

                        

                        

                            
❓ 是否需要定时刷新 Token？

                            
建议设置定时任务，在 token 过期前 1-2 天自动刷新，确保服务的连续性。

func callAPIWithRetry(ctx context.Context, maxRetries int) error {
    for i := 0; i < maxRetries; i++ {
        resp, _, err := client.AdvertiserApi.Get(ctx, req)
        
        if err == nil && resp.Code == 0 {
            return nil  // 成功
        }
        
        // 判断是否需要重试
        if shouldRetry(err, resp) {
            backoff := time.Duration(i+1) * time.Second
            log.Printf("重试 %d/%d, 等待 %v", i+1, maxRetries, backoff)
            time.Sleep(backoff)
            continue
        }
        
        return err  // 不可重试错误
    }
    return fmt.Errorf("超过最大重试次数")
}

func shouldRetry(err error, resp *models.Response) bool {
    // 网络超时错误
    if err != nil && strings.Contains(err.Error(), "timeout") {
        return true
    }
    // 限流错误 (40004)
    if resp != nil && resp.Code == 40004 {
        return true
    }
    return false
}

import "sync"

// 使用 Worker Pool 控制并发
type WorkerPool struct {
    maxWorkers int
    tasks      chan func()
    wg         sync.WaitGroup
}

func NewWorkerPool(maxWorkers int) *WorkerPool {
    pool := &WorkerPool{
        maxWorkers: maxWorkers,
        tasks:      make(chan func(), 100),
    }
    
    // 启动 worker
    for i := 0; i < maxWorkers; i++ {
        go pool.worker()
    }
    
    return pool
}

func (p *WorkerPool) worker() {
    for task := range p.tasks {
        task()
        p.wg.Done()
    }
}

func (p *WorkerPool) Submit(task func()) {
    p.wg.Add(1)
    p.tasks <- task
}

func (p *WorkerPool) Wait() {
    p.wg.Wait()
}

type TokenManager struct {
    accessToken  string
    refreshToken string
    expiresAt    time.Time
    mu           sync.RWMutex
}

func (tm *TokenManager) GetToken(ctx context.Context) (string, error) {
    tm.mu.RLock()
    defer tm.mu.RUnlock()
    
    // 检查 Token 是否过期
    if time.Now().After(tm.expiresAt.Add(-10 * time.Minute)) {
        // 提前 10 分钟刷新
        return tm.refreshAccessToken(ctx)
    }
    
    return tm.accessToken, nil
}

func (tm *TokenManager) refreshAccessToken(ctx context.Context) (string, error) {
    tm.mu.Lock()
    defer tm.mu.Unlock()
    
    // 调用刷新接口
    resp, _, err := client.OAuthApi.RefreshToken(ctx, &api.RefreshTokenRequest{
        RefreshToken: tm.refreshToken,
    })
    
    if err != nil {
        return "", err
    }
    
    tm.accessToken = resp.Data.AccessToken
    tm.refreshToken = resp.Data.RefreshToken
    tm.expiresAt = time.Now().Add(time.Duration(resp.Data.ExpiresIn) * time.Second)
    
    return tm.accessToken, nil
}
